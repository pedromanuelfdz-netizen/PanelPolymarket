""" main.py — Polymarket Football Latency Bot + API endpoint """ import asyncio import logging import sys import json from datetime import datetime from pathlib import Path from aiohttp import web Path("logs").mkdir(exist_ok=True) logging.basicConfig( level=logging.INFO, format="%(asctime)s | %(name)-14s | %(levelname)s | %(message)s", handlers=[ logging.StreamHandler(sys.stdout), logging.FileHandler(f"logs/bot_{datetime.now().strftime('%Y%m%d')}.log"), ] ) log = logging.getLogger("main") from modules.polymarket import PolymarketClient import config.settings as cfg poly = PolymarketClient() markets_cache = [] CORS = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Methods": "GET, OPTIONS", "Access-Control-Allow-Headers": "*", } GAMMA_API = "https://gamma-api.polymarket.com" async def handle_options(request): return web.Response(status=200, headers=CORS) async def handle_health(request): return web.json_response({ "status": "ok", "markets_loaded": len(markets_cache), "dry_run": cfg.DRY_RUN, }, headers=CORS) async def handle_debug(request): sample = markets_cache[:30] result = [{"question": m.get("question", ""), "title": m.get("title", "")} for m in sample] return web.json_response({"total": len(markets_cache), "sample": result}, headers=CORS) async def fetch_event_markets(session, slug): """Fetch all markets for a given event slug. Returns (title, markets).""" try: async with session.get(f"{GAMMA_API}/events", params={"slug": slug}) as r: if r.status != 200: return None, [] data = await r.json() events = data if isinstance(data, list) else data.get("events", []) if not events: return None, [] ev = events[0] markets = [] for m in ev.get("markets", []): question = m.get("question") or ev.get("title") or "" raw_tokens = m.get("clobTokenIds") or [] if isinstance(raw_tokens, str): try: raw_tokens = json.loads(raw_tokens) except: raw_tokens = [] if not raw_tokens: raw_tokens = [ t.get("token_id") for t in m.get("tokens", []) if t.get("token_id") ] markets.append({ "id": m.get("id"), "question": question, "token_ids": raw_tokens, "volume": m.get("volume"), "outcomes": m.get("outcomes"), }) return ev.get("title", slug), markets except Exception as e: log.warning(f"fetch_event_markets({slug}) error: {e}") return None, [] async def handle_event(request): url = request.rel_url.query.get("url", "").strip() slug = request.rel_url.query.get("slug", "").strip() if url and not slug: parts = [p for p in url.rstrip("/").split("/") if p] slug = parts[-1] if parts else "" if not slug: return web.json_response({"error": "provide url or slug"}, headers=CORS, status=400) try: session = await poly._get_session() # Fetch main event and more-markets event in parallel title1, markets1 = await fetch_event_markets(session, slug) title2, markets2 = await fetch_event_markets(session, slug + "-more-markets") all_markets = markets1 + markets2 event_title = title1 or title2 or slug # Deduplicate by market id seen = set() unique = [] for m in all_markets: mid = m.get("id") if mid not in seen: seen.add(mid) unique.append(m) if not unique: return web.json_response( {"error": f"No markets found for: {slug}"}, headers=CORS, status=404 ) log.info(f"Event: {event_title} — {len(unique)} markets ({len(markets1)} main + {len(markets2)} more)") return web.json_response({ "event": event_title, "slug": slug, "markets": unique, }, headers=CORS) except Exception as e: log.error(f"Event lookup error: {e}") return web.json_response({"error": str(e)}, headers=CORS, status=500) async def handle_search(request): home = request.rel_url.query.get("home", "").strip() away = request.rel_url.query.get("away", "").strip() query = " ".join(filter(None, [home, away])) if not query: return web.json_response( {"error": "provide at least one team"}, headers=CORS, status=400 ) try: words = [w for w in query.lower().split() if len(w) > 2] results = [] for m in markets_cache: q = (m.get("question") or m.get("title") or "").lower() if any(w in q for w in words): raw_tokens = m.get("clobTokenIds") or [] if isinstance(raw_tokens, str): try: raw_tokens = json.loads(raw_tokens) except: raw_tokens = [] results.append({ "id": m.get("id"), "question": m.get("question") or m.get("title"), "token_ids": raw_tokens, "volume": m.get("volume"), }) seen = set() unique = [] for r in results: if r["question"] not in seen: seen.add(r["question"]) unique.append(r) return web.json_response({"markets": unique[:8]}, headers=CORS) except Exception as e: return web.json_response({"error": str(e)}, headers=CORS, status=500) async def handle_orderbook(request): token_id = request.rel_url.query.get("token_id", "").strip() if not token_id: return web.json_response( {"error": "token_id required"}, headers=CORS, status=400 ) try: book = await poly.get_orderbook(token_id) if not book: return web.json_response({"error": "no book"}, headers=CORS, status=404) return web.json_response({ "best_ask": poly.best_ask(book), "best_bid": poly.best_bid(book), "spread": poly.spread(book), "bids": book.get("bids", [])[:6], "asks": book.get("asks", [])[:6], }, headers=CORS) except Exception as e: return web.json_response({"error": str(e)}, headers=CORS, status=500) async def refresh_markets(): global markets_cache while True: try: markets_cache = await poly.get_football_markets(limit=200) log.info(f"Markets refreshed: {len(markets_cache)}") except Exception as e: log.error(f"Market refresh failed: {e}") await asyncio.sleep(300) async def main(): log.info("=" * 60) log.info("POLYMARKET BOT + API — STARTING") log.info(f"Mode: {'DRY RUN' if cfg.DRY_RUN else 'LIVE'}") log.info("=" * 60) try: markets_cache.extend(await poly.get_football_markets(limit=200)) log.info(f"Loaded {len(markets_cache)} markets") except Exception as e: log.error(f"Initial market load failed: {e}") app = web.Application() app.router.add_route("OPTIONS", "/{path_info:.*}", handle_options) app.router.add_get("/health", handle_health) app.router.add_get("/debug", handle_debug) app.router.add_get("/event", handle_event) app.router.add_get("/search", handle_search) app.router.add_get("/orderbook", handle_orderbook) runner = web.AppRunner(app) await runner.setup() site = web.TCPSite(runner, "0.0.0.0", 8080) await site.start() log.info("API server running on port 8080") asyncio.create_task(refresh_markets()) while True: await asyncio.sleep(60) if __name__ == "__main__": try: asyncio.run(main()) except KeyboardInterrupt: log.info("Bot stopped")
